# Base58

## Install
`npm i --save @dwlib/base58`

## Usage
```javascript
// CJS
const Base58 = require('@dwlib/base58');
const Base58Decode = require('@dwlib/base58/decode');
const Base58DecodeBigInt = require('@dwlib/base58/decodeBigInt');
const Base58DecodeInt = require('@dwlib/base58/decodeInt');
const Base58DecodeInto = require('@dwlib/base58/decodeInto');
const Base58Encode = require('@dwlib/base58/encode');
const Base58EncodeBigInt = require('@dwlib/base58/encodeBigInt');
const Base58EncodeInt = require('@dwlib/base58/encodeInt');
const Base58IsValid = require('@dwlib/base58/isValid');
const Base58Validate = require('@dwlib/base58/validate');
// ESM
import Base58 from '@dwlib/base58';
import Base58Decode from '@dwlib/base58/decode';
import Base58DecodeBigInt from '@dwlib/base58/decodeBigInt';
import Base58DecodeInt from '@dwlib/base58/decodeInt';
import Base58DecodeInto from '@dwlib/base58/decodeInto';
import Base58Encode from '@dwlib/base58/encode';
import Base58EncodeBigInt from '@dwlib/base58/encodeBigInt';
import Base58EncodeInt from '@dwlib/base58/encodeInt';
import Base58IsValid from '@dwlib/base58/isValid';
import Base58Validate from '@dwlib/base58/validate';
```
